CREATE TABLE PRODUTOS ( 
 IDPRODUTO NUMBER PRIMARY KEY, 
 NOMEPRODUTO VARCHAR(70) NOT NULL, 
 CARACTERISTICAS VARCHAR(100) NOT NULL, 
 PRECOMEDIO NUMBER, 
 LOGOTIPO 	VARCHAR(200) NOT NULL, 
 ATIVO 	INT, 
 IDTIPO NUMBER, 
 CONSTRAINT FK_IDTIPO FOREIGN KEY (IDTIPO) REFERENCES TIPOPRODUTO(IDTIPO) 
); 

 

CREATE SEQUENCE PRODUTO_IDPRODUTO_SEQ;  

CREATE OR REPLACE TRIGGER TR_SEQ_CONTATO BEFORE INSERT ON PRODUTOS FOR EACH ROW  
	BEGIN 
		SELECT PRODUTO_IDPRODUTO_SEQ.NEXTVAL INTO :new.IDPRODUTO  FROM dual; 
END;
